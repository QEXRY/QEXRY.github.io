#include<iostream>
using namespace std;
int leapy(int n){
    if(n%400==0) return 1;
    if(n%4==0 && n%100) return 1;
    return 0;
}

int month(int n,int m){
    if(n==1 || n==3 || n==5 || n==7 || n==8 || n==10 || n==12) return 31;
    if(n==2 && leapy(m)) return 29;
    if(n==2) return 28;
    if(n==4 || n==6 || n==9 || n==11) return 30;
}

int wrongdate(int x,int y,int z){
if(y<=0 || z<=0) return 1;
if(y>12) return 1;
if(z>month(y,x)) return 1;
}

int montoday(int n,int m){
if(n==0) return 0;
if(n==1) return month(1,m);
else return montoday(n-1,m)+month(n,m);
}

int datetoday(int x,int y,int z){
int sum=0;
for(int i=1920;i<x;i++)
    {
        if(leapy(i)) sum+=366;
        else sum+=365;
    }
    sum+=montoday(y,x)+z;
    return sum;
}

int main(){
int y,m,d;
cin>>y>>m>>d;
while(1){
if(wrongdate(y,m,d)) {cout<<"日期不合法";break;}
int ans=datetoday(y,m,d)-datetoday(1946,10,5);
if(ans>=0) {cout<<"吉林大学已经成立了"<<ans<<"天";break;}
else {cout<<"距吉林大学成立还有"<<-ans<<"天";break;}
}
return 0;
}
